## 3190. [뱀](https://www.acmicpc.net/problem/3190)


``` java
import java.io.*;
import java.util.LinkedList;
import java.util.Queue;
import java.util.StringTokenizer;

public class Main{
	static int N, K, L, second=0;
	static int[][] Board;
	static int[][] pos = {{0,1},{1,0},{0,-1},{-1,0}};
	static int[] count;
	static char[] direc;
	public static void main(String[] args) throws Exception{
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		N = Integer.parseInt(br.readLine());
		Board = new int[N+1][N+1];
		StringTokenizer st;
		
		K = Integer.parseInt(br.readLine());
		for(int i=0; i<K; i++) {
			st = new StringTokenizer(br.readLine());
			Board[Integer.parseInt(st.nextToken())][Integer.parseInt(st.nextToken())] = 2;
		}
		
		L = Integer.parseInt(br.readLine());
		count = new int[L];
		direc = new char[L];
		for(int i=0; i<L; i++) {
			st = new StringTokenizer(br.readLine());
			count[i] = Integer.parseInt(st.nextToken());
			direc[i] = st.nextToken().charAt(0);
		}
		for(int i=L-1; i>0; i--) {
			count[i] = count[i]-count[i-1];
		}
		Move(1,1);
		System.out.println(second);
	}
	static class XY{
		int x;
		int y;
		public XY(int x, int y) {
			super();
			this.x = x;
			this.y = y;
		}
		
	}
	private static void Move(int r, int c) {
		Queue<XY> q = new LinkedList<>();
		boolean isBreak=false;
		Board[r][c] = 1;
		q.offer(new XY(r,c));
		int d=0;
		Top:
		for(int i=0; i<L; i++) {			
			int cnt = count[i];
			for(int j=0; j<cnt; j++) {
				second++;
				int nr = r+pos[d][0];
				int nc = c+pos[d][1];
				
				if(Stop(nr,nc)) {
					isBreak = true;
					break Top;
				}
				
				if(Board[nr][nc]==0) {//빈칸일때
					Board[nr][nc] = 1;
					q.offer(new XY(nr,nc));
					XY xy = q.poll();
					Board[xy.x][xy.y] = 0;
				}else if(Board[nr][nc]==2) {//사과 먹을때
					Board[nr][nc] = 1;
					q.offer(new XY(nr,nc));
				}				
				r = nr;
				c = nc;
			}
			if(direc[i]=='L') {
				if(d==0)
					d=3;
				else
					d-=1;
			}else {
				if(d==3)
					d=0;
				else
					d+=1;
			}
		}
		if(!isBreak) {
			while(true){
				second++;
				int nr = r+pos[d][0];
				int nc = c+pos[d][1];
				if(Stop(nr,nc)) {
					isBreak = true;
					break;
				}
				r = nr;
				c = nc;
			}
		}
	}
	
	private static boolean Stop(int r, int c) {
		if(r<=0 || c<=0 || r>N || c>N || Board[r][c]==1) {
			return true;
		}	
		return false;
	}
}
```
